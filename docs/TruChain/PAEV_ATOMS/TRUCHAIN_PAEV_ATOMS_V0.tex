\section{Purpose}
This document is the audit notebook for the \textbf{PAEV Atom Ladder}: a staged, falsifiable test program that moves from
\textbf{simulation invariants} to \textbf{physical wave-field demonstrations} and then to \textbf{atom-adjacent proxies}
(without overclaiming).

\textbf{Primary goal:} demonstrate that the PAEV ``wave program'' grammar yields reproducible, measurable control effects,
and that those effects survive transition from simulation to hardware proxies.

\textbf{Guardrails (MUST):}
\begin{itemize}
  \item The ladder MUST avoid biological / wetlab claims.
  \item The ladder MUST avoid claims of atomic manipulation unless measurements explicitly support them.
  \item Each non-simulation rung MUST be backed by immutable artifacts (raw logs + SHA256) and a measured results JSON.
\end{itemize}

\section{Ladder overview}

\subsection*{A --- Simulation (prove grammar + invariants)}
\begin{itemize}
  \item \textbf{A1 Topology Gate Causality (Freeze + g sweep)}\\
    Claim: \texttt{topo\_gate01} is a real control input; Freeze pins topology deterministically.\\
    Pass: Freeze cuts topology-change-rate $\ge 90\%$ AND at least one stability metric monotonic with $g$ ($\rho \ge 0.7$).

  \item \textbf{A2 Resonant Addressing (Node Frequency IDs)}\\
    Claim: nodes are selectively addressable by wave tokens $(\omega,\phi,A)$, not global gain.\\
    Pass: selectivity $S \ge 5$ and crosstalk $X \le 0.2$ for $\ge 80\%$ targets across seeds.

  \item \textbf{A3.1 Chiral Handshake (Protocol Compatibility Gate)}\\
    Claim: chirality acts as a header that gates coupling (match $\rightarrow$ couple, mismatch $\rightarrow$ suppress).\\
    Pass: $\Delta\text{coupling} = C_{\text{match}} - C_{\text{mismatch}} > 3\sigma$ across trials; decode-error jumps on mismatch.

  \item \textbf{A4 Wormhole / k\_link vs Distance (Adjacency beats geometry)}\\
    Claim: informational adjacency changes coordination independent of Euclidean distance (in-sim).\\
    Pass: distance sensitivity (slope) drops $\ge 50\%$ with \texttt{k\_link} enabled (equal energy budget).
\end{itemize}

\subsection*{B --- Physical wave-field (prove ``wave program writes geometry'')}
\begin{itemize}
  \item \textbf{B1 Chladni Pattern Compiler (geometry-from-program)}\\
    Pass: similarity $\ge 0.8$ on $\ge 3$ distinct target modes; robustness drop $\le 0.1$ under small perturbations.

  \item \textbf{B2 Trap Node Control (deterministic routing)}\\
    Pass: RMSE $< 5\%$ travel range; dropout $< 10\%$ at moderate speed.

  \item \textbf{B3 Information Channel (symbols over waves)}\\
    Pass: SER $< 5\%$ at nominal SNR with graceful degradation curve.
\end{itemize}

\subsection*{C --- Atom-adjacent proxy (cash-out without overclaiming)}
\begin{itemize}
  \item \textbf{C1 EM Resonant Addressing Proxy (LC ``atoms'' + routing)}\\
    Pass: selectivity $\ge 5$; topology-rule success $\ge 90\%$; stable under detuning sweep.

  \item \textbf{C2 Two-level-system emulator (optional but strong)}\\
    Pass: fidelity above baseline and fidelity $\ge 0.80$.

  \item \textbf{C4 Multi-node Structural Hold (``acoustic molecule'')}\\
    Pass: stable hold for $T$ seconds with bounded shape error; recovery $\ge 80\%$; beats independent-traps baseline.
\end{itemize}
