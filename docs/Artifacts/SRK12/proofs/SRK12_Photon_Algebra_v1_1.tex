\section{Core Operators}
\subsection{Superposition}
\[
\Eval(X \superpose Y) = \Eval(X) + \Eval(Y).
\]

\subsection{Phase-Parameterized Fusion $\fuse{\phi}$}
\[
\Eval(X \fuse{\phi} Y) = \Eval(X) + e^{i\phi}\Eval(Y).
